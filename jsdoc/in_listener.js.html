 34 //This function changes the color of the rating stars based on the star the user clicked on.
 35 $(document).ready(function(){
 36 	$(".ratingStar").click(function(){
 37 
 38 	var id = "";
 39 	var starCount = 1;
 40 	var on = true;
 41 	var stars = ["star1", "star2", "star3", "star4", "star5"];
 42 	
 43 	// Activate and Deactivate stars based on user click
 44 	id = $(this).attr("id");
 45 	for(var i = 0; i < stars.length; i++){
 46 		if(stars[i] == id){
 47 			$("#"+stars[i]).addClass('activeStar');
 48 			on = false;
 49 		} else {
 50 			if(on){
 51 				$("#"+stars[i]).addClass('activeStar');
 52 				starCount++;
 53 			} else {
 54 				$("#"+stars[i]).removeClass('activeStar');
 55 			}
 56 		}
 57 	}
 58 	
 59 	$("#ratingValue").val(starCount);
 60 	
 61 	});
 62 })
