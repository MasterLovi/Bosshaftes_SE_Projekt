100 /**
101  * Shifts the route list to the left, if the user clicked a button
102  * @returns NONE
103  */
104 function toureListLeftShift() {
105 	// Disables the button until the animation is done. Prevents confusing
106 	// result if the button was clicked too often
107 	$("#leftArrow").unbind("click");
108 
109 	$("#tourList").animate({
110 		"margin-left" : "+=170" // 170 is the width of a list element 
111 	}, "slow", function() {
112 		var x = $("#tourList").css("margin-left");
113 		var offSet = Number(x.substring(0, x.length - 2));
114 
115 		// Checking if the offSet of the List is 0. This is the default
116 		// situation. If the page loads the offset will be 0
117 		if (offSet >= 0) {
118 			if ($("#leftArrow").css("display") != "none") {
119 				$("#leftArrow").css("display", "none");
120 			}
121 		}
122 		$("#rightArrow").css("display", "block");
123 
124 		// Rebinds the function to the button, so if it is clicked the next time
125 		// the animation will start again
126 		$("#leftArrow").bind("click", toureListLeftShift);
127 	});
128 }

130 /**
131  * Shifts the route to the right, if the user clicked a button
132  * @returns NONE
133  */
134 function toureListRightShift() {
135 
136 	// Disables the button until the animation is done. Prevents confusing
137 	// result if the button was clicked too often
138 	$("#rightArrow").unbind("click");
139 
140 	$("#tourList").animate({
141 		"margin-left" : "-=170" // 170 is the size of a tour list element
142 	}, "slow", function() {

144 		var toureListSize = ($("#tourList li").length - 1) * 170; 
145 
146 		width = $("#tours").css("width");

150 		var offSet = Number(width.substring(0, width.length - 2));
151 
152 		// Offset is always negative

157 		if (toureListSize <= viewPortSize - offSet) {
158 			if ($("#rightArrow").css("display") != "none") {
159 				$("#rightArrow").css("display", "none");

161 		}

165 		// the animation will start again
166 		$("#rightArrow").bind("click", toureListRightShift);

358 /**
359  * Sends the stops information from a specific rounte to the mapquest direction API and draw the route on the map.
360  * @param type {String} - Contains the information where to search for the selected route.
361  * @returns NONE
362  */
363 function calculateRoute(type) {
364 	var tour = $("#tourIdOnPanle").val();
365 	var tourObj;
366 	
367 	// makes sure that if a new route is calculated, the old route will be removed frist.
368 	removeCurrentRoute();
369 	
370 	// Checks where to search for the route. If the user call the function from the manage route propup, then search in userRoutes for it.
371 	if (type == "global") {
372 		$.each(globalRoutes, function(i, v) {
373 			if (v.id == tour) {
374 				tourObj = v;
375 				return;
376 			}
377 		});
378 	} else if (type == "user") {
379 		$.each(userRoutes, function(i, v) {
380 			if (v.id == tour) {
381 				tourObj = v;
382 				return;
383 			}
384 		});
385 	}
386 
387 	// load the basic structure for the required json object
388 	var routeObj = getRoutingJsonStructure();
389 
390 	$.each(tourObj.stops, function(i, v) {
391 		var newElement = {
392 			type : "s",
393 			latLng : {
394 				lat : v.latitude,
395 				lng : v.longitude
396 			}
397 		};
398 		routeObj.locations.push(newElement)
399 	});
400 
401 	// this option makes sure that the api return a route that is optimized for pedestrian
402 	routeObj.options = {
403 		routeType : "pedestrian"
404 	};
405 
406 	// evokes the route function of the api and draws the route
407 	L.mapquest.directions().route(routeObj);
408 	
409 }

1104 /**
1105  * 
1106  * @param startTime
1107  * @param endTime
1108  * @returns
1109  */
1110 function addTimes (startTime, endTime) {
1111 	  var times = [ 0, 0, 0 ]
1112 	  var max = times.length
1113 
1114 	  var a = (startTime || '').split(':')
1115 	  var b = (endTime || '').split(':')
1116 
1117 	  // normalize time values
1118 	  for (var i = 0; i < max; i++) {
1119 	    a[i] = isNaN(parseInt(a[i])) ? 0 : parseInt(a[i])
1120 	    b[i] = isNaN(parseInt(b[i])) ? 0 : parseInt(b[i])
1121 	  }
1122 
1123 	  // store time values
1124 	  for (var i = 0; i < max; i++) {
1125 	    times[i] = a[i] + b[i]
1126 	  }
1127 
1128 	  var hours = times[0]
1129 	  var minutes = times[1]
1130 	  var seconds = times[2]
1131 
1132 	  if (seconds >= 60) {
1133 	    var m = (seconds / 60) << 0
1134 	    minutes += m
1135 	    seconds -= 60 * m
1136 	  }
1137 
1138 	  if (minutes >= 60) {
1139 	    var h = (minutes / 60) << 0
1140 	    hours += h
1141 	    minutes -= 60 * h
1142 	  }
1143 
1144 	  return ('0' + hours).slice(-2) + ':' + ('0' + minutes).slice(-2) + ':' + ('0' + seconds).slice(-2)
1145 	}
