514 /**
515  * Deletes a existing feedback from a route or location
516  * @param type {String} - Contains the information if it is a location or route feedback
517  * @param id {Integer} - Contains the id of the route of location 
518  * @param feedbackId {Integer} - Contains the id of the feedback that should be deleted
519  * @returns NONE
520  */
521 function deleteFeedback(type, id, feedbackId) {
522 	var json;
523 	var feedback;
524 	var locationId;
525 	var indexOfFeedback;
526 	var tourIndex;
527 	
528 	// Get right locatin or route based on type
529 	if (type == "Location") {
530 		json = globalLayer.getLayer(id).info;
531 	} else {
532 		$.each(globalRoutes, function(i,v){
533 			if (v.id == id) {
534 				json = v;
535 				tourIndex = i;
536 				return;
537 			}
538 		});
539 	}
540 	
541 	// Get the location id form the extracted location / route
542 	locationId = json.id;
543 	
544 	// Find the feedback in the location or route object that should be deleted
545 	$.each(json.feedback, function(i,v) {
546 		if(v.id == feedbackId){
547 			feedback = v;
548 			indexOfFeedback = i;
549 			return;
550 		}
551 	});
552 	
553 	// Bring the data in a backend compatible format
554 	var jsonArray = [feedback];
555 	
556 	// Make the ajax call
557 	$.ajax({
558 		url: "FeedbackServlet",
559 		type: "POST",
560 		data: {
561 			type: type,
562 			id: locationId,
563 			operation: "delete",
564 			json: JSON.stringify(jsonArray)
565 		}, 
566 		success: function(response) {
567 			
568 			if (type == "Location") {
569 				// Deletes the recently deleted feedback from the globalLayer
570 				globalLayer.getLayer(id).info.feedback.splice(indexOfFeedback, 1);
571 				// recalculate the average rating 
572 				globalLayer.getLayer(id).info.avgRating = calculateAvgRating(globalLayer.getLayer(id).info);
573 			} else {
574 				// Deletes the recently deleted feedback from the globalRoutes object
575 				globalRoutes[tourIndex].feedback.splice(indexOfFeedback, 1);
576 				// Recalculate the average rating 
577 				globalRoutes[tourIndex].avgRating = calculateAvgRating(globalRoutes[tourIndex]);
578 				// Reload the globalRoutes object and reload the info panel
579 				getRoute($("#currentAction").val());
580 				loadRouteToPanel(id);
581 			}
582 			// Reloads the feedback
583 			loadFeedbackToPopup({id: id, type: type});
584 			
585 			// Reload the global layer data
586 			refreshLayerData($("#currentAction").val());
587 			
588 			// Status 
589 			sendStatusMessage("Feedback erfolgreich gelöscht.", "green");
590 		},
591 		error: function(error) {
592 			console.log(error);
593 		}
594 	});
595 }

799 /**
800  * Remove a location from a user route. Route can not contain less than one location
801  * @param locationId {Integer} - Contains the leaflet_id of the location that should be removed
802  * @param routeId {Integer} - Contains the route id of the route the location should be removed from
803  * @returns NONE
804  */
805 function removeFromRoute(locationId, routeId) {
806 	var removeIndex;
807 	var route;
808 	var location;
809 	var pTimeCalculate;
810 	
811 	// Find the proper user route 
812 	$.each(userRoutes, function(i,v) {
813 		if(v.id == routeId) {
814 			route = v;
815 			return;
816 		}
817 	});
818 	
819 	// Find the proper location in the stops array
820 	$.each(route.stops, function(i,v) {
821 		if (v.id == locationId) {
822 			removeIndex = i;
823 		}
824 	});
825 	
826 	// Check if that stop is the only stop in the array
827 	if (route.numberOfStops-1 == 0) {
828 		$("#popupError").html("Die Route darf nicht leer sein. Sie können den Punkt nicht löschen!");
829 		cancelDeletionOrChange('route');
830 		return;
831 	}
832 	
833 	// Remove the location from the stops array
834 	route.stops.splice(removeIndex, 1); //Removes the element on index 'removeIndex' in the Array
835 	route.numberOfStops = route.numberOfStops-1;
836 	// Recalculate travel time
837 	pTimeCalculate = calculateTraveltime(route);
838 	
839 	// Stops code execution until time was calculated 
840 	pTimeCalculate.then(function(time) {
841 		
842 		if (route.type == "Kultur"){
843 			route.time.time = calculateTotalTravelTime(route, time);
844 		} else {
845 			route.time.time = time;
846 		}
847 		
848 		// Bring the data in a backend compatible format
849 		var jsonArray = [route];
850 		
851 		$.ajax({
852 			url: "RouteServlet",
853 			type: "POST",
854 			data: {
855 				operation: "update",
856 				json: JSON.stringify(jsonArray)
857 			},
858 			success: function(response) {
859 				// Reload route information + Status
860 				changeRouteInformation(routeId);
861 				sendStatusMessage("Punkt erfolgreich aus Route gelöscht.", "green");
862 			},
863 			error: function(error) {
864 				console.log(error);
865 			}
866 		});
867 	});
